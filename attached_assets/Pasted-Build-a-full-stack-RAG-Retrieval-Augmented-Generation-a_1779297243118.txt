from langchain.memory import ConversationBufferMemory
from app.core.embedder import get_store
from app.config import settings
from app.models import ChatMessage

def build_rag_chain(doc_id: str):
    """
    Build a ConversationalRetrievalChain:
    - Get FAISS store for doc_id
    - Create retriever with k=settings.retrieval_k
    - Create ChatOpenAI with settings.chat_model, temperature=0
    - Return the chain (no memory — we pass history manually)
    """

def query_document(
    doc_id: str,
    question: str,
    history: list[ChatMessage]
) -> dict:
    """
    Run a RAG query:
    - Build the chain
    - Convert history list to LangChain format: 
      list of (human_message, ai_message) tuples
    - Call chain with question and chat_history
    - Return dict:
      {
        "answer": str,

      }

## Key behaviours to implement correctly

RAG Pipeline flow (get this exactly right):
1. User uploads PDF
2. PyMuPDF extracts raw text page by page
3. RecursiveCharacterTextSplitter splits into overlapping chunks
4. OpenAI embeddings convert each chunk to a vector
5. FAISS stores all vectors in memory indexed by doc_id
6. User asks a question
7. Question is embedded using same model
8. FAISS finds top-K most similar chunks (cosine similarity)
9. Retrieved chunks + question + chat history sent to GPT
10. GPT answers using ONLY the retrieved context
11. Answer + source excerpts returned to frontend

System prompt for the LLM (use this exactly):
"""
You are DocuBOT, a helpful assistant that answers questions about 
uploaded documents. Answer ONLY based on the provided context. 
If the answer is not in the context, say "I couldn't find that 
information in the document." Be concise and accurate.
"""

Error handling requirements:
- Empty PDF: return 400 "PDF appears to be empty or unreadable"
- PDF too large: return 400 "File exceeds 10MB limit"  
- Non-PDF file: return 400 "Only PDF files are accepted"
- doc_id not found: return 404 "Document not found. Please upload again."
- OpenAI API error: return 500 "AI service temporarily unavailable"
